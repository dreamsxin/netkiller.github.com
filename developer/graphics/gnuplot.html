$ cat test.log
8:00 506.877

$ cat test.gnuplot
set terminal png truecolor size 800,250
set output "test.png"
set autoscale
set xdata time
set timefmt "%H:%M"
set style data lines
set xlabel "time per day"
set ylabel "Mbps"
set title "Apache Traffic"
set grid
plot "test.log" using 1:2 title "Hit"

$ gnuplot test.gnuplot
